public int climbStairs(int n) {
    if (n <= 2)
        return n;

    int i = 1, j = 2, cur = 0;
    for(int k = 2; k < n ; k ++) {
        cur = i + j;
        i = j;
        j = cur;
    }
    return cur;
}
