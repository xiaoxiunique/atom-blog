public void recursive(int level, int param) {
	// terminator 
    if (level > MAX_LEVEL) {
        // process return 
        return;
    }
    
    // process current logic
    process(level, param);
    
    // drill down
    recursive(level: level + 1, newParam);
    
    // restore current status
}
